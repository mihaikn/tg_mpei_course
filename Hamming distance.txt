                int k=0;
        while(x!=0 && y!=0)
        {
            if(x%2!=y%2)
                k++;
            x=x/2;
            y=y/2;
        }
        while(x!=0)
        {
            if(x%2==1)
                k++;
            x=x/2;
        }
        while(y!=0)
        {
            if(y%2==1)
                k++;
            y=y/2;
        }
        return k;
